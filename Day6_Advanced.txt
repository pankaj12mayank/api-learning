## ✅ 3. services/task_service.py

```python id="fc4b7j"
from database.db import task_db

def create_task_service(task):
    task_db.append(task)
    return task

def get_all_tasks_service():
    return task_db

def get_task_service(task_id):
    for t in task_db:
        if t.id == task_id:
            return t
    return None

def update_task_service(task_id, updated_task):
    for index, t in enumerate(task_db):
        if t.id == task_id:
            task_db[index] = updated_task
            return updated_task
    return None

def delete_task_service(task_id):
    for index, t in enumerate(task_db):
        if t.id == task_id:
            task_db.pop(index)
            return True
    return False
```

---

## ✅ 4. routes/task_routes.py

```python id="8w3r2n"
from fastapi import APIRouter
from models.task_model import Task
from services.task_service import *

router = APIRouter()

@router.post("/tasks")
def create_task(task: Task):
    return create_task_service(task)

@router.get("/tasks")
def get_tasks():
    return get_all_tasks_service()

@router.get("/tasks/{task_id}")
def get_task(task_id: int):
    task = get_task_service(task_id)
    if task:
        return task
    return {"error": "Task not found"}

@router.put("/tasks/{task_id}")
def update_task(task_id: int, task: Task):
    updated = update_task_service(task_id, task)
    if updated:
        return updated
    return {"error": "Task not found"}

@router.delete("/tasks/{task_id}")
def delete_task(task_id: int):
    deleted = delete_task_service(task_id)
    if deleted:
        return {"message": "Task deleted"}
    return {"error": "Task not found"}
```
